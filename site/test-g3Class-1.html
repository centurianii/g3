   constructor: function(options){
      var myClass = 'A';
      // if inheritance level > 1 and you want object storage:
      //g3[myClass].add(this.defaults.name, this);
      this.instance.prop_A = null,
      this.init(options);
      this.instance.newBuild[myClass] = false;
   },
   
   prototype: {
      init: function(options){
         var debug = {};
         
         // 1. call functions
         debug['switch'] = this.switch(options, 'A');
         debug['getNodes'] = this.getNodes(options);
         debug['build'] = this.build();

         return this;
      },
      build: function(){
         this.instance.prop_A = 3;
      },
      toString: function(){
         return '[Object g3.A]';
      }
   }
